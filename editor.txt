#include <stdio.h>
#include <errno.h>
#include <string.h>

extern int errno ;

int main () {
   FILE *fp;

   fp = fopen("file.txt", "r");
   if( fp == NULL ) {
      fprintf(stderr, "Value of errno: %d", errno);
      fprintf(stderr, "Error opening file: %s", strerror(errno));
   } else {
      fclose(fp);
   }

   return(0);
}
